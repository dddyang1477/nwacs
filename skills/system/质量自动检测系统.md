### 2.1 检测项目
```python
class BasicQualityDetector:
    """基础质量检测器"""

    def detect(self, content):
        results = {
            'grammar_errors': self.check_grammar(content),
            'spelling_errors': self.check_spelling(content),
            'punctuation_issues': self.check_punctuation(content),
            'clarity_issues': self.check_clarity(content),
            'logic_issues': self.check_logic(content)
        }

        # 综合评分
        overall_score = self.calculate_score(results)

        return {
            'passed': overall_score >= 85,
            'score': overall_score,
            'issues': results,
            'suggestions': self.generate_suggestions(results)
        }

    def check_grammar(self, content):
        """语法错误检测"""
        issues = []

        # 检测主谓不一致
        subject_verb = self.detect_subject_verb_mismatch(content)

        # 检测句子成分残缺
        incomplete = self.detect_incomplete_sentences(content)

        # 检测修饰语位置不当
        modifier = self.detect_misplaced_modifiers(content)

        return {
            'detected': len(subject_verb) + len(incomplete) + len(modifier),
            'details': subject_verb + incomplete + modifier,
            'error_rate': self.calculate_error_rate(content)
        }

    def check_logic(self, content):
        """逻辑连贯性检测"""
        issues = []

        # 检测时间线矛盾
        timeline = self.detect_timeline_contradictions(content)

        # 检测前后矛盾
        contradiction = self.detect_internal_contradictions(content)

        # 检测因果不合理
        causation = self.detect_causation_issues(content)

        return {
            'logic_score': self.calculate_logic_score(content),
            'issues': timeline + contradiction + causation
        }
```

### 2.2 常见问题与修复
```
基础质量问题修复指南：

1. 语法错误
   问题：主谓不一致、句子成分残缺
   修复：自动语法修正 + 人工确认

2. 表达不清晰
   问题：句意模糊、指代不明
   修复：重新表述 + 添加明确指代

3. 逻辑不通
   问题：时间线矛盾、因果不合理
   修复：回溯上下文 + 逻辑修正
```

### 3.1 AI特征识别
```python
class AITraceDetector:
    """AI痕迹检测器"""

    # AI特征模式库
    AI_PATTERNS = {
        'perfect_sentence': [
            # 完美句式（缺乏自然瑕疵）
            r'她\s+缓缓地\s+轻轻地\s+温柔地\s+轻轻地',
            r'他\s+静静地\s+默默地\s+静静地\s+沉默着',
            r'天空\s+湛蓝无比\s+阳光\s+明媚灿烂\s+微风\s+轻柔拂过'
        ],

        'template_structure': [
            # 模板化结构
            r'只见那[^\s]+\s+如同\s+[^\s]+一般',
            r'此时此刻\s+[^\s]+\s+心中\s+[^\s]+',
            r'不得不说\s+[^\s]+\s+实在是\s+[^\s]+'
        ],

        'emotion_amplification': [
            # 过度情感放大
            r'心中\s+无比\s+[^\s]+',
            r'脸上\s+露出\s+极其\s+[^\s]+',
            r'整个人\s+都\s+感觉\s+[^\s]+'
        ],

        'logical_absence': [
            # 逻辑断层
            r'突然\s+就\s+[^\s]+了',
            r'然后\s+就\s+[^\s]+了',
            r'没想到\s+竟然\s+[^\s]+'
        ],

        'over_description': [
            # 过度描写
            r'她的\s+长发\s+如同\s+瀑布\s+般\s+倾泻而下',
            r'他的\s+眼神\s+深邃\s+而\s+神秘\s+仿佛\s+能够\s+洞察\s+一切'
        ]
    }

    def detect(self, content):
        """综合AI痕迹检测"""

        results = {
            'perfect_sentence_count': self.count_pattern(content, 'perfect_sentence'),
            'template_structure_count': self.count_pattern(content, 'template_structure'),
            'emotion_amplification_count': self.count_pattern(content, 'emotion_amplification'),
            'logical_absence_count': self.count_pattern(content, 'logical_absence'),
            'over_description_count': self.count_pattern(content, 'over_description')
        }

        # AI特征比例
        total_sentences = len(content.split('。'))
        ai_feature_count = sum(results.values())
        ai_feature_rate = ai_feature_count / total_sentences if total_sentences > 0 else 0

        # 人类化评分
        humanization_score = self.calculate_humanization_score(content, results)

        return {
            'ai_feature_rate': ai_feature_rate,
            'humanization_score': humanization_score,
            'passed': ai_feature_rate < 0.15 and humanization_score >= 75,
            'issues': results,
            'risk_level': self.assess_risk_level(ai_feature_rate)
        }
```

class RhythmAnalyzer:
    """节奏爽点分析器"""

    def analyze(self, chapter_content, previous_chapters=None):
        """综合节奏分析"""

        # 1. 节奏流畅度分析
        rhythm_flow = self.analyze_rhythm_flow(chapter_content)

        # 2. 爽点密度检测
        shuang_density = self.detect_shuang_points(chapter_content)

        # 3. 情绪曲线分析
        emotion_curve = self.analyze_emotion_curve(chapter_content)

        # 4. 高潮间隔检测
        climax_interval = self.detect_climax_interval(
            chapter_content,
            previous_chapters
        )

        # 5. 读者期待管理
        expectation = self.analyze_reader_expectation(chapter_content)

        return {
            'rhythm_flow': rhythm_flow,
            'shuang_density': shuang_density,
            'emotion_curve': emotion_curve,
            'climax_interval': climax_interval,
            'reader_expectation': expectation,
            'overall_score': self.calculate_rhythm_score(
                rhythm_flow,
                shuang_density,
                emotion_curve,
                climax_interval
            )
        }

    def detect_shuang_points(self, content):
        """爽点密度检测"""
        shuang_types = {
            '升级爽': ['突破', '升级', '进阶', '提升', '觉醒'],
            '打脸爽': ['打脸', '碾压', '震惊', '后悔', '目瞪口呆'],
            '获得爽': ['获得', '得到', '拥有', '收获', '意外收获'],
            '守护爽': ['保护', '守护', '拯救', '挺身而出'],
            '复仇爽': ['复仇', '报复', '讨回', '雪耻']
        }

        detected_shuang = {}

        for shuang_type, keywords in shuang_types.items():
            count = sum(content.count(kw) for kw in keywords)
            detected_shuang[shuang_type] = count

        total_shuang = sum(detected_shuang.values())

        return {
            'total_count': total_shuang,
            'by_type': detected_shuang,
            'density_per_1000_chars': (total_shuang / len(content)) * 1000,
            'assessment': self.assess_density(total_shuang, len(content))
        }
```

    def analyze_emotion_curve(self, content):
        """情绪曲线分析"""

        # 将章节分成5个段落
        segments = self.split_into_segments(content, 5)

        emotion_scores = []

        for segment in segments:
            score = self.calculate_emotion_score(segment)
            emotion_scores.append(score)

        # 分析曲线特征
        curve_features = {
            'scores': emotion_scores,
            'variation': max(emotion_scores) - min(emotion_scores),
            'trend': 'increasing' if emotion_scores[-1] > emotion_scores[0] else 'stable',
            'peak_position': emotion_scores.index(max(emotion_scores)),
            'curve_type': self.classify_curve_type(emotion_scores)
        }

        return {
            'curve': emotion_scores,
            'features': curve_features,
            'health_score': self.assess_curve_health(curve_features)
        }

    def classify_curve_type(self, scores):
        """曲线类型分类"""
        if scores[-1] > scores[0] and max(scores) == scores[-1]:
            return '一路上升型'  # 最佳结尾
        elif max(scores) in scores[2:4] and scores[-1] < max(scores):
            return '高潮在中后段'  # 良好
        elif max(scores) == scores[0]:
            return '开头高潮型'  # 需要优化
        else:
            return '波动型'  # 需要调整

class ComplianceDetector:
    """平台合规检测器"""

    # 违规词库（示例，实际库更大）
    VIOLATION_WORDS = {
        'politically_sensitive': ['敏感政治词汇...'],
        'vulgar': ['低俗词汇...'],
        'discriminatory': ['歧视性词汇...'],
        'violent': ['过度暴力描写...'],
        'advertising': ['广告法禁用词...']
    }

    def detect_violations(self, content):
        """违规词检测"""

        violations = []

        # 政治敏感词
        political = self.check_words(content, self.VIOLATION_WORDS['politically_sensitive'])
        if political:
            violations.append({
                'type': '政治敏感',
                'words': political,
                'severity': 'high',
                'action': '必须修改'
            })

        # 低俗词汇
        vulgar = self.check_words(content, self.VIOLATION_WORDS['vulgar'])
        if vulgar:
            violations.append({
                'type': '低俗用语',
                'words': vulgar,
                'severity': 'medium',
                'action': '建议修改'
            })

        # 广告法禁用词
        advertising = self.check_advertising_words(content)
        if advertising:
            violations.append({
                'type': '广告法违规',
                'words': advertising,
                'severity': 'medium',
                'action': '必须修改'
            })

        return {
            'total_violations': len(violations),
            'high_severity': len([v for v in violations if v['severity'] == 'high']),
            'violations': violations,
            'compliance_score': self.calculate_compliance_score(violations),
            'can_publish': len([v for v in violations if v['severity'] == 'high']) == 0
        }
```
